# Milestone 2 -- authentication
# bcrypt is pinned explicitly (not just via passlib[bcrypt]) because
# passlib 1.7.4's backend-detection code is incompatible with bcrypt>=4.1
# (it raises "password cannot be longer than 72 bytes" during its own
# internal self-test, unrelated to any password a user actually enters).
# passlib has been unmaintained since 2020 and never shipped a fix; 4.0.1
# is the newest bcrypt release confirmed compatible with it.
passlib[bcrypt]==1.7.4
bcrypt==4.0.1
python-jose[cryptography]==3.3.0
email-validator==2.2.0

# Milestone 3 -- document upload & ingestion
# PyMuPDF: PDF text extraction (app/services/extraction.py). python-multipart:
# FastAPI/Starlette requires it to parse multipart/form-data file uploads --
# without it, UploadFile routes raise at request time, not at import time,
# so it's easy to miss until someone actually uploads a file.
PyMuPDF==1.24.10
python-multipart==0.0.12
# httpx: app/services/embeddings.py imports it directly at module level for
# OpenAIEmbeddingProvider. It was already present transitively via
# qdrant-client, but a module that imports a package directly should
# declare it directly rather than rely on another package's transitive
# dependency, which could change or be dropped without notice.
httpx==0.27.2
